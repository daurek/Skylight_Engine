110  void Scene::LoadComponents(rapidxml::xml_node<>* components, Entity & entity)
111  {
112  Module::ModuleMap & factories = Module::get_module_map();
113 
114  for (rapidxml::xml_node<>* component_node = components->first_node(); component_node; component_node = component_node->next_sibling())
115  {
116  if (std::string(component_node->name()) == "component")
117  {
118  std::string component_name = component_node->first_attribute()->value();
119 
120  std::shared_ptr<Module> module = nullptr;
121 
122  // Creating factories since static objects do not get created in libs
123  if (factories.count(component_name) == 0)
124  {
125  if (component_name == "render3d")
126  factories[component_name] = &RenderModule::factory;
127  else if (component_name == "transform")
128  factories[component_name] = &TransformModule::factory;
129  else if (component_name == "audio")
130  factories[component_name] = &AudioModule::factory;
131  else if (component_name == "camera")
132  factories[component_name] = &CameraModule::factory;
133  else if (component_name == "light")
134  factories[component_name] = &LightModule::factory;
135  }
136 
137  // Does that module not exist on our scene
138  if (modules.count(component_name) == 0)
139  {
140  modules[component_name] = factories[component_name]->CreateModule(*this);
141  module = modules[component_name];
142  SDL_Log("\n [_______Created module_______]");
143  std::string log_module = " " + component_name;

147  // If it exists then get it
148  else

151  }
152
