23  void InputTask::Run()
24  {
25  SDL_Event event;
26 
27  // Get the event
28  while (SDL_PollEvent(&event) > 0)
29  {
30  // Check type
31  switch (event.type)
32  {
33  case SDL_KEYDOWN:
34  {
35  auto sdl_keycode = keyboard_mapping.find(event.key.keysym.sym);
36 
37  if (sdl_keycode != keyboard_mapping.end())
38  {
39  // Send Key Pressed message with Key Code
40  Message message("key_pressed");
41  message.AddProperty("key_code", sdl_keycode->second);
42  MessageHandler::instance().multicast(message);
43  }
44  break;
45  }
46  case SDL_MOUSEMOTION:
47  {
48  int x, y;

52  message.AddProperty("position_y", y);

54 
55  break;
56  }
57 
58  case SDL_QUIT:
59  {
60  Kernel::instance().Stop();
